function onEdit(e) {
  if (!e || !e.range) {
    return;
  }

  var hoja = e.range.getSheet();

  if (hoja.getName() !== "Inventario") {
    if (hoja.getName() === "MISE AN PLACE") {
      ProcesarEntradaSalidaMiseAnPlace_(e);
    }

    if (hoja.getName() === "Salida(s)" || hoja.getName() === "Decomiso") {
      ActualizarMensajeExcedeInventarioFormulario_(hoja);
    }

    return;
  }

  var filaInicial = e.range.getRow();
  var columnaInicial = e.range.getColumn();
  var filaFinal = filaInicial + e.range.getNumRows() - 1;
  var columnaFinal = columnaInicial + e.range.getNumColumns() - 1;
  var columnasInventario = ObtenerColumnasInventario_(hoja);

  if (
    filaFinal < 4 ||
    columnaInicial > columnasInventario.producto ||
    columnaFinal < columnasInventario.producto
  ) {
    return;
  }

  var primeraFilaInventario = Math.max(filaInicial, 4);
  var cantidadFilas = filaFinal - primeraFilaInventario + 1;

  if (cantidadFilas === 1 && EsRenglonSeparadorInventario_(hoja, primeraFilaInventario, columnasInventario)) {
    return;
  }

  ReiniciarMovimientosFilasInventario_(hoja, primeraFilaInventario, cantidadFilas, columnasInventario);
  ActualizarFormulasConteoRenglonesInventario_(hoja);
  SincronizarProductosInventarioSilencioso_();
}

function ProcesarEntradaSalidaMiseAnPlace_(e) {
  var hoja = e.range.getSheet();
  var filaInicial = e.range.getRow();
  var columnaInicial = e.range.getColumn();
  var filaFinal = filaInicial + e.range.getNumRows() - 1;
  var columnaFinal = columnaInicial + e.range.getNumColumns() - 1;

  if (filaFinal < 5 || columnaFinal < 4 || columnaInicial > 5) {
    return;
  }

  var primeraFila = Math.max(filaInicial, 5);
  var cantidadFilas = filaFinal - primeraFila + 1;
  var rangoDatos = hoja.getRange(primeraFila, 2, cantidadFilas, 5);
  var datos = rangoDatos.getValues();
  var limpiarMovimientos = false;

  for (var i = 0; i < datos.length; i++) {
    var producto = String(datos[i][0] || "").trim();

    if (producto === "") {
      continue;
    }

    var existencia = parseFloat(datos[i][1]);
    var entrada = parseFloat(datos[i][2]);
    var salida = parseFloat(datos[i][3]);

    if (isNaN(existencia)) {
      existencia = 0;
    }

    if (isNaN(entrada)) {
      entrada = 0;
    }

    if (isNaN(salida)) {
      salida = 0;
    }

    if (entrada === 0 && salida === 0) {
      continue;
    }

    var nuevaExistencia = existencia + entrada - salida;

    if (nuevaExistencia < 0) {
      nuevaExistencia = 0;
    }

    datos[i][1] = nuevaExistencia;
    datos[i][2] = "";
    datos[i][3] = "";
    datos[i][4] = nuevaExistencia;
    limpiarMovimientos = true;
  }

  if (limpiarMovimientos) {
    rangoDatos.setValues(datos);
  }
}

function GenerarListadoCompra() {
  var ui = SpreadsheetApp.getUi();
  var respuesta = ui.prompt(
    "Listado de compra",
    'Escriba el stock a usar:\n\n1 = LUNES, MARTES, MIERCOLES, JUEVES\n2 = VIERNES, SABADO, DOMINGO',
    ui.ButtonSet.OK_CANCEL
  );

  if (respuesta.getSelectedButton() !== ui.Button.OK) {
    return;
  }

  var opcion = String(respuesta.getResponseText() || "").trim().toLowerCase();
  var tipoStock = "";

  if (opcion === "1" || opcion === "l" || opcion === "lmj" || opcion === "l,m,m,j" || opcion.indexOf("lunes") !== -1) {
    tipoStock = "lunes_jueves";
  } else if (opcion === "2" || opcion === "v" || opcion === "vsd" || opcion === "v,s,d" || opcion.indexOf("viernes") !== -1) {
    tipoStock = "viernes_domingo";
  } else {
    ui.alert("Opcion no valida", "Debe escribir 1 o 2.", ui.ButtonSet.OK);
    return;
  }

  var resultado = CrearListadoCompra_(tipoStock);

  ui.alert(
    "Listado de compra generado",
    "Se generaron " + resultado.total + " productos para comprar.\n\nHoja: " + resultado.hoja,
    ui.ButtonSet.OK
  );
}

function CrearListadoCompra_(tipoStock) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var inventario = ObtenerHojaObligatoria_(ss, "Inventario");
  var columnas = ObtenerColumnasInventario_(inventario);
  var ultimaFila = inventario.getLastRow();
  var nombreHoja = "LISTADO DE COMPRA";
  var hojaListado = ss.getSheetByName(nombreHoja);

  if (!hojaListado) {
    hojaListado = ss.insertSheet(nombreHoja);
  }

  hojaListado.clear();

  var tituloStock = tipoStock === "lunes_jueves"
    ? "LUNES, MARTES, MIERCOLES, JUEVES"
    : "VIERNES, SABADO, DOMINGO";
  var columnaStock = tipoStock === "lunes_jueves"
    ? columnas.stockLunesJueves
    : columnas.stockViernesDomingo;
  var columnasInformacion = ObtenerColumnasInformacionProductoInventario_(inventario, columnas);
  var encabezadosInformacion = ObtenerEncabezadosColumnasInventario_(inventario, columnasInformacion);
  var encabezadosListado = encabezadosInformacion.concat([
    "Stock requerido",
    "Existencia actual",
    "Comprar"
  ]);
  var totalColumnasListado = encabezadosListado.length;

  hojaListado.getRange(1, 1, 1, totalColumnasListado).merge();
  hojaListado.getRange(1, 1).setValue("LISTADO DE COMPRA - " + tituloStock);
  hojaListado.getRange(2, 1).setValue("Generado");
  hojaListado.getRange(2, 2).setValue(new Date());
  hojaListado.getRange(4, 1, 1, totalColumnasListado).setValues([encabezadosListado]);

  if (ultimaFila < 4) {
    return { hoja: nombreHoja, total: 0 };
  }

  var datos = inventario.getRange(4, 1, ultimaFila - 3, inventario.getLastColumn()).getDisplayValues();
  var compras = [];

  for (var i = 0; i < datos.length; i++) {
    var producto = String(datos[i][columnas.producto - 1] || "").trim();
    var informacionProducto = [];
    var stock = ConvertirNumero_(datos[i][columnaStock - 1]);
    var existencia = ConvertirNumero_(datos[i][columnas.existencia - 1]);

    if (producto === "" || isNaN(stock) || stock <= 0) {
      continue;
    }

    if (isNaN(existencia)) {
      existencia = 0;
    }

    if (existencia < stock) {
      for (var j = 0; j < columnasInformacion.length; j++) {
        informacionProducto.push(datos[i][columnasInformacion[j] - 1]);
      }

      compras.push(informacionProducto.concat([
        stock,
        existencia,
        stock - existencia
      ]));
    }
  }

  if (compras.length > 0) {
    hojaListado.getRange(5, 1, compras.length, totalColumnasListado).setValues(compras);
  } else {
    hojaListado.getRange(5, 1).setValue("No hay productos por debajo del stock seleccionado.");
  }

  FormatearListadoCompra_(hojaListado, compras.length, totalColumnasListado, encabezadosInformacion.length + 1);

  return { hoja: nombreHoja, total: compras.length };
}

function FormatearListadoCompra_(hoja, cantidadFilas, totalColumnas, primeraColumnaNumerica) {
  var ultimaFilaListado = Math.max(4, cantidadFilas + 4);

  hoja.getRange(1, 1, ultimaFilaListado, totalColumnas)
    .setFontSize(18)
    .setHorizontalAlignment("center")
    .setVerticalAlignment("middle");

  hoja.getRange(1, 1, 1, totalColumnas)
    .setBackground("#1f7a3a")
    .setFontColor("#ffffff")
    .setFontWeight("bold")
    .setHorizontalAlignment("center");

  hoja.getRange(4, 1, 1, totalColumnas)
    .setBackground("#d9ead3")
    .setFontWeight("bold")
    .setHorizontalAlignment("center");

  hoja.setFrozenRows(4);
  hoja.setRowHeights(1, ultimaFilaListado, 34);
  hoja.autoResizeColumns(1, totalColumnas);

  if (cantidadFilas > 0) {
    hoja.getRange(5, primeraColumnaNumerica, cantidadFilas, 3).setNumberFormat("0.##");
  }
}

function MigrarListadoCompraADrive() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var hojaListado = ObtenerHojaObligatoria_(ss, "LISTADO DE COMPRA");
  var ultimaFila = hojaListado.getLastRow();
  var ultimaColumna = hojaListado.getLastColumn();

  if (ultimaFila < 5 || ultimaColumna < 1) {
    SpreadsheetApp.getUi().alert(
      "Listado vacio",
      'Primero debe generar informacion en la hoja "LISTADO DE COMPRA".',
      SpreadsheetApp.getUi().ButtonSet.OK
    );
    return;
  }

  var fecha = new Date();
  var fechaTexto = FormatearFechaCompraGenerada_(fecha);
  var horaTexto = Utilities.formatDate(fecha, ss.getSpreadsheetTimeZone(), "HH-mm-ss");
  var nombreArchivo = "LISTADO DE COMPRA - " + fechaTexto + " - " + horaTexto;
  var archivoDestino = SpreadsheetApp.create(nombreArchivo);
  var ssDestino = SpreadsheetApp.openById(archivoDestino.getId());
  var hojaCopiada = hojaListado.copyTo(ssDestino);
  var hojasDestino = ssDestino.getSheets();
  var archivoDrive = DriveApp.getFileById(archivoDestino.getId());

  hojaCopiada.setName("LISTADO DE COMPRA");

  for (var i = 0; i < hojasDestino.length; i++) {
    if (hojasDestino[i].getSheetId() !== hojaCopiada.getSheetId()) {
      ssDestino.deleteSheet(hojasDestino[i]);
    }
  }

  CopiarDimensionesHoja_(hojaListado, hojaCopiada);
  ssDestino.setActiveSheet(hojaCopiada);
  archivoDrive.addEditors(CORREOS_DESTINO_LISTADOS_COMPRA);
  SpreadsheetApp.flush();

  EnviarCorreoListadoCompraMigrado_(CORREOS_DESTINO_LISTADOS_COMPRA, nombreArchivo, archivoDestino);
  LimpiarListadoCompraDespuesDeMigrar_(hojaListado);

  SpreadsheetApp.getUi().alert(
    "Listado enviado",
    "Se envio el LISTADO DE COMPRA por correo y se limpio la hoja original.\n\n" +
      "Correos: " + CORREOS_DESTINO_LISTADOS_COMPRA.join(", ") + "\n" +
      "Archivo: " + nombreArchivo + "\n\n" +
      "Enlace:\n" + archivoDestino.getUrl(),
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}

function EnviarCorreoListadoCompraMigrado_(correoDestino, nombreArchivo, archivoDestino) {
  MailApp.sendEmail({
    to: correoDestino.join(","),
    subject: nombreArchivo,
    body:
      "Se genero un nuevo LISTADO DE COMPRA.\n\n" +
      "Archivo: " + nombreArchivo + "\n" +
      "Enlace:\n" + archivoDestino.getUrl()
  });
}

function FormatearFechaCompraGenerada_(fecha) {
  var meses = ["Ene", "Feb", "Mar", "Abr", "May", "Jun", "Jul", "Ago", "Sep", "Oct", "Nov", "Dic"];

  return fecha.getDate() + "/" + meses[fecha.getMonth()] + "/" + fecha.getFullYear();
}

function CopiarDimensionesHoja_(hojaOrigen, hojaDestino) {
  var ultimaFila = hojaOrigen.getLastRow();
  var ultimaColumna = hojaOrigen.getLastColumn();

  for (var columna = 1; columna <= ultimaColumna; columna++) {
    hojaDestino.setColumnWidth(columna, hojaOrigen.getColumnWidth(columna));
  }

  for (var fila = 1; fila <= ultimaFila; fila++) {
    hojaDestino.setRowHeight(fila, hojaOrigen.getRowHeight(fila));
  }
}

function LimpiarListadoCompraDespuesDeMigrar_(hojaListado) {
  hojaListado.clear();
  hojaListado.setFrozenRows(0);
}

function SincronizarProductosInventario_(silencioso) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var inventario = ObtenerHojaObligatoria_(ss, "Inventario");
  var productos = ObtenerProductosInventario_(inventario);
  var rangoProductos = ActualizarHojaProductosInventario_(ss, productos);
  var hojasFormulario = ["Entrada(s)", "Salida(s)", "Decomiso"];
  var hojasActualizadas = [];

  for (var i = 0; i < hojasFormulario.length; i++) {
    if (AplicarListaProductosAFormulario_(ss, hojasFormulario[i], rangoProductos)) {
      hojasActualizadas.push(hojasFormulario[i]);
    }
  }

  if (!silencioso && hojasActualizadas.length === 0) {
    throw new Error('No se encontro la etiqueta "Producto" en Entrada(s), Salida(s) o Decomiso.');
  }

  return {
    total: productos.length,
    hojasActualizadas: hojasActualizadas
  };
}

function ObtenerProductosInventario_(inventario, columnas) {
  var ultimaFila = inventario.getLastRow();

  if (ultimaFila < 4) {
    return [];
  }

  columnas = columnas || ObtenerColumnasInventario_(inventario);
  var ultimaColumna = inventario.getLastColumn();
  var datos = inventario.getRange(4, 1, ultimaFila - 3, ultimaColumna).getDisplayValues();
  var productos = [];
  var vistos = {};

  for (var i = 0; i < datos.length; i++) {
    var filaInventario = i + 4;
    var producto = String(datos[i][columnas.producto - 1] || "").trim();
    var clave = NormalizarTexto_(producto);

    if (producto === "" || vistos[clave] || EsRenglonSeparadorInventario_(inventario, filaInventario, columnas)) {
      continue;
    }

    vistos[clave] = true;
    productos.push({
      producto: producto,
      medida: datos[i][columnas.medida - 1] || "",
      suplidor: datos[i][columnas.suplidor - 1] || "",
      existencia: datos[i][columnas.existencia - 1] || ""
    });
  }

  productos.sort(function (a, b) {
    return a.producto.localeCompare(b.producto);
  });

  return productos;
}

function ActualizarHojaProductosInventario_(ss, productos) {
  var hojaProductos = ss.getSheetByName(NOMBRE_HOJA_PRODUCTOS_INVENTARIO);

  if (!hojaProductos) {
    hojaProductos = ss.insertSheet(NOMBRE_HOJA_PRODUCTOS_INVENTARIO);
  }

  hojaProductos.clear();
  hojaProductos.getRange(1, 1, 1, 4).setValues([["Producto", "Medida", "Suplidor", "Existencia"]]);

  if (productos.length === 0) {
    hojaProductos.getRange(2, 1, 1, 4).setValues([["", "", "", ""]]);
  } else {
    var valores = productos.map(function (item) {
      return [item.producto, item.medida, item.suplidor, item.existencia];
    });

    hojaProductos.getRange(2, 1, valores.length, 4).setValues(valores);
  }

  hojaProductos.autoResizeColumns(1, 4);
  hojaProductos.hideSheet();

  return hojaProductos.getRange(2, 1, Math.max(productos.length, 1), 1);
}

function AplicarListaProductosAFormulario_(ss, nombreHoja, rangoProductos) {
  var hoja = ss.getSheetByName(nombreHoja);

  if (!hoja) {
    return false;
  }

  var rangoProducto = ObtenerPrimeraCeldaEntradaEditablePorEtiqueta_(hoja, "Producto");

  if (!rangoProducto) {
    return false;
  }

  var validacion = SpreadsheetApp.newDataValidation()
    .requireValueInRange(rangoProductos, true)
    .setAllowInvalid(false)
    .build();

  rangoProducto.setDataValidation(validacion);

  return true;
}

function ObtenerPrimeraCeldaEntradaEditablePorEtiqueta_(hoja, etiqueta) {
  var celdaEtiqueta = BuscarCeldaPorEtiqueta_(hoja, etiqueta);

  if (!celdaEtiqueta) {
    return null;
  }

  var candidatos = ObtenerCeldasDeEntrada_(hoja, celdaEtiqueta);

  for (var i = 0; i < candidatos.length; i++) {
    var rango = ObtenerRangoCompletoSiEstaCombinado_(candidatos[i]);

    if (!rango.getFormula()) {
      return rango;
    }
  }

  return null;
}

function RegistrarMovimientoInventario_(config) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var formulario = ObtenerHojaObligatoria_(ss, config.hojaFormulario);
  var registro = ObtenerHojaObligatoria_(ss, config.hojaRegistro);
  var inventario = ObtenerHojaObligatoria_(ss, "Inventario");
  var columnas = ObtenerColumnasInventario_(inventario);

  var producto = ObtenerValorPorEtiqueta_(formulario, "Producto");
  var cantidad = parseFloat(String(ObtenerValorPorEtiqueta_(formulario, "Cantidad")).replace(",", "."));

  if (producto === "") {
    throw new Error('Debe seleccionar un producto en "' + config.hojaFormulario + '".');
  }

  if (isNaN(cantidad) || cantidad <= 0) {
    throw new Error('Debe ingresar una cantidad valida mayor a cero en "' + config.hojaFormulario + '".');
  }

  var infoProducto = BuscarProductoInventario_(inventario, producto, columnas);

  if (!infoProducto) {
    throw new Error('El producto "' + producto + '" no existe en la hoja "Inventario".');
  }

  if ((config.tipo === "salida" || config.tipo === "decomiso") && cantidad > infoProducto.existencia) {
    throw new Error(
      "La cantidad excede la existencia actual.\n\n" +
      "Producto: " + infoProducto.producto + "\n" +
      "Existencia actual: " + infoProducto.existencia + "\n" +
      "Cantidad solicitada: " + cantidad
    );
  }

  var filaRegistro = Math.max(registro.getLastRow() + 1, 3);

  AcumularMovimientoInventario_(inventario, infoProducto, config.tipo, cantidad, columnas);

  registro.getRange(filaRegistro, 1, 1, 5).setValues([[
    new Date(),
    infoProducto.numero,
    infoProducto.medida,
    infoProducto.producto,
    cantidad
  ]]);

  if (config.limpiarDespues) {
    LimpiarFormularioMovimiento_(config.hojaFormulario, true);
  }

  SpreadsheetApp.flush();

  SpreadsheetApp.getUi().alert(
    "Registro guardado",
    "Se guardo correctamente.\n\nProducto: " + infoProducto.producto + "\nCantidad: " + cantidad,
    SpreadsheetApp.getUi().ButtonSet.OK
  );
}

function LimpiarFormularioMovimiento_(nombreHoja, silencioso) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var hoja = ObtenerHojaObligatoria_(ss, nombreHoja);

  LimpiarValorPorEtiqueta_(hoja, "Producto");
  LimpiarValorPorEtiqueta_(hoja, "Cantidad");

  if (nombreHoja === "Salida(s)" || nombreHoja === "Decomiso") {
    ObtenerCeldaMensajeExcedeInventario_(hoja).clearContent();
  }

  SpreadsheetApp.flush();

  if (!silencioso) {
    SpreadsheetApp.getUi().alert(
      "Formulario limpio",
      'Se limpiaron Producto y Cantidad en "' + nombreHoja + '".',
      SpreadsheetApp.getUi().ButtonSet.OK
    );
  }
}

function ObtenerBloqueoDocumento_() {
  var lock = LockService.getDocumentLock();

  if (!lock.tryLock(5000)) {
    throw new Error(
      "Hay otro registro ejecutandose en este momento.\n\n" +
      "Espere unos segundos y vuelva a intentarlo. Si el problema continua, recargue la hoja."
    );
  }

  return lock;
}

function LiberarBloqueo_(lock) {
  if (lock) {
    lock.releaseLock();
  }
}

function ActualizarMensajeExcedeInventarioFormulario_(hoja) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var inventario = ObtenerHojaObligatoria_(ss, "Inventario");
  var celdaMensaje = ObtenerCeldaMensajeExcedeInventario_(hoja);
  var producto = ObtenerValorPorEtiquetaSinError_(hoja, "Producto");
  var cantidadTexto = ObtenerValorPorEtiquetaSinError_(hoja, "Cantidad");
  var cantidad = parseFloat(String(cantidadTexto || "").replace(",", "."));

  if (producto === "" || isNaN(cantidad) || cantidad <= 0) {
    celdaMensaje.clearContent();
    return;
  }

  var infoProducto = BuscarProductoInventario_(inventario, producto);

  if (!infoProducto || cantidad > infoProducto.existencia) {
    celdaMensaje.setValue("❌ Excede inventario ❌");
  } else {
    celdaMensaje.clearContent();
  }
}

function ObtenerValorPorEtiquetaSinError_(hoja, etiqueta) {
  try {
    return ObtenerValorPorEtiqueta_(hoja, etiqueta);
  } catch (error) {
    return "";
  }
}

function ObtenerCeldaMensajeExcedeInventario_(hoja) {
  var celdaCantidad = BuscarCeldaPorEtiqueta_(hoja, "Cantidad");

  if (!celdaCantidad) {
    return hoja.getRange("B8");
  }

  var filaMensaje = celdaCantidad.getRow() + 1;
  var columnaMensaje = celdaCantidad.getColumn() + 1;
  var rangoMensaje = hoja.getRange(filaMensaje, columnaMensaje);

  return ObtenerRangoCompletoSiEstaCombinado_(rangoMensaje);
}

function AcumularMovimientoInventario_(inventario, infoProducto, tipo, cantidad, columnas) {
  columnas = columnas || ObtenerColumnasInventario_(inventario);
  var columnaMovimiento = columnas[tipo];

  if (!columnaMovimiento) {
    throw new Error('No se encontro la columna de "' + tipo + '" en la hoja "Inventario".');
  }

  var celdaMovimiento = inventario.getRange(infoProducto.fila, columnaMovimiento);

  if (!celdaMovimiento.getFormula()) {
    var valorActual = parseFloat(celdaMovimiento.getValue());

    if (isNaN(valorActual)) {
      valorActual = 0;
    }

    celdaMovimiento.setValue(valorActual + cantidad);
  }

  ActualizarFormulaInventarioAutomatico_(inventario, infoProducto.fila, columnas);
}

function ActualizarFormulaInventarioAutomatico_(inventario, fila, columnas) {
  var celdaExistencia = inventario.getRange(fila, columnas.existencia);

  celdaExistencia.setFormulaR1C1(ObtenerFormulaInventarioAutomaticoR1C1_(columnas));
}

function ObtenerFormulaInventarioAutomaticoR1C1_(columnas) {
  var offsetEntrada = columnas.entrada - columnas.existencia;
  var offsetSalida = columnas.salida - columnas.existencia;
  var offsetDecomiso = columnas.decomiso - columnas.existencia;

  return "=RC[" + offsetEntrada + "]-RC[" + offsetSalida + "]-RC[" + offsetDecomiso + "]";
}

function ReiniciarMovimientosInventario_(inventario, filaInicial, cantidadFilas) {
  var columnas = ObtenerColumnasInventario_(inventario);
  var formulaInventario = ObtenerFormulaInventarioAutomaticoR1C1_(columnas);
  var ceros = [];

  for (var i = 0; i < cantidadFilas; i++) {
    ceros.push([0, 0, 0]);
  }

  if (columnas.salida === columnas.entrada + 1 && columnas.decomiso === columnas.entrada + 2) {
    inventario.getRange(filaInicial, columnas.entrada, cantidadFilas, 3).setValues(ceros);
  } else {
    inventario.getRange(filaInicial, columnas.entrada, cantidadFilas, 1).setValue(0);
    inventario.getRange(filaInicial, columnas.salida, cantidadFilas, 1).setValue(0);
    inventario.getRange(filaInicial, columnas.decomiso, cantidadFilas, 1).setValue(0);
  }

  inventario.getRange(filaInicial, columnas.existencia, cantidadFilas, 1).setFormulaR1C1(formulaInventario);
}

function ReiniciarMovimientosFilasInventario_(inventario, filaInicial, cantidadFilas, columnas) {
  columnas = columnas || ObtenerColumnasInventario_(inventario);
  var inicioBloque = null;
  var cantidadBloque = 0;

  for (var i = 0; i < cantidadFilas; i++) {
    var fila = filaInicial + i;

    if (EsRenglonSeparadorInventario_(inventario, fila, columnas)) {
      if (inicioBloque !== null) {
        ReiniciarMovimientosInventario_(inventario, inicioBloque, cantidadBloque);
        inicioBloque = null;
        cantidadBloque = 0;
      }

      continue;
    }

    if (inicioBloque === null) {
      inicioBloque = fila;
      cantidadBloque = 1;
    } else {
      cantidadBloque++;
    }
  }

  if (inicioBloque !== null) {
    ReiniciarMovimientosInventario_(inventario, inicioBloque, cantidadBloque);
  }
}

function LimpiarFilasVaciasInventario_(inventario) {
  var ultimaFila = inventario.getLastRow();

  if (ultimaFila < 4) {
    return 0;
  }

  var columnas = ObtenerColumnasInventario_(inventario);
  var datosBase = inventario.getRange(4, 1, ultimaFila - 3, 3).getDisplayValues();
  var filasALimpiar = [];

  for (var i = 0; i < datosBase.length; i++) {
    var filaInventario = i + 4;

    if (EsRenglonSeparadorInventario_(inventario, filaInventario, columnas)) {
      continue;
    }

    var filaIncompleta = datosBase[i].some(function (valor) {
      return String(valor || "").trim() === "";
    });

    if (filaIncompleta) {
      filasALimpiar.push(filaInventario);
    }
  }

  ReiniciarFilasEnBloques_(inventario, filasALimpiar);

  return filasALimpiar.length;
}

function ObtenerFilaNuevoRenglonInventario_(inventario) {
  var ultimaFila = Math.max(inventario.getLastRow(), 4);
  var cantidadFilas = ultimaFila - 3;
  var columnas = ObtenerColumnasInventario_(inventario);
  var columnaInicial = columnas.producto;
  var columnaFinal = Math.max(columnas.stockViernesDomingo, columnas.producto);
  var datosBase = inventario.getRange(4, columnaInicial, cantidadFilas, columnaFinal - columnaInicial + 1).getDisplayValues();
  var ultimaFilaConDatos = 0;

  for (var i = 0; i < datosBase.length; i++) {
    var tieneDatos = datosBase[i].some(function (valor) {
      return String(valor || "").trim() !== "";
    });

    if (tieneDatos) {
      ultimaFilaConDatos = i + 4;
    }
  }

  if (ultimaFilaConDatos === 0) {
    return 4;
  }

  return ultimaFilaConDatos + 1;
}

function PrepararFilaRenglonInventario_(inventario, fila, nombreRenglon) {
  var columnas = ObtenerColumnasInventario_(inventario);

  if (fila > inventario.getMaxRows()) {
    inventario.insertRowsAfter(inventario.getMaxRows(), fila - inventario.getMaxRows());
  } else if (FilaTieneContenidoOFormula_(inventario, fila)) {
    inventario.insertRowsBefore(fila, 1);
  }

  var ultimaColumna = Math.max(inventario.getLastColumn(), columnas.existencia);
  var rangoFila = inventario.getRange(fila, 1, 1, ultimaColumna);
  var columnaFinalRenglon = Math.max(columnas.stockViernesDomingo, columnas.medida, columnas.producto);
  var rangoRenglon = inventario.getRange(fila, columnas.producto, 1, columnaFinalRenglon - columnas.producto + 1);

  rangoFila.breakApart();
  rangoFila.clearContent();

  rangoRenglon.merge();
  rangoRenglon
    .setValue(nombreRenglon)
    .setBackground("#ea9999")
    .setFontWeight("bold")
    .setHorizontalAlignment("center")
    .setVerticalAlignment("middle");

  inventario.getRange(fila, 1).clearContent();
  inventario.getRange(fila, columnas.entrada, 1, columnas.existencia - columnas.entrada + 1).clearContent();
  ActualizarFormulasConteoRenglonesInventario_(inventario);
}

function FilaTieneContenidoOFormula_(hoja, fila) {
  var ultimaColumna = hoja.getLastColumn();
  var rango = hoja.getRange(fila, 1, 1, ultimaColumna);
  var valores = rango.getDisplayValues()[0];
  var formulas = rango.getFormulas()[0];

  for (var i = 0; i < ultimaColumna; i++) {
    if (String(valores[i] || "").trim() !== "" || String(formulas[i] || "").trim() !== "") {
      return true;
    }
  }

  return false;
}

function ActualizarFormulasConteoRenglonesInventario_(inventario) {
  var columnas = ObtenerColumnasInventario_(inventario);
  var ultimaFila = Math.max(inventario.getLastRow(), 4);
  var filasSeparadoras = [];

  inventario.getRange(4, columnas.numero, inventario.getMaxRows() - 3, 1).clearContent();

  for (var fila = 4; fila <= ultimaFila; fila++) {
    if (EsRenglonSeparadorInventario_(inventario, fila, columnas)) {
      filasSeparadoras.push(fila);
    }
  }

  if (filasSeparadoras.length === 0) {
    ActualizarConteoBloqueInventario_(inventario, 4, ultimaFila, columnas);
    return;
  }

  for (var i = 0; i < filasSeparadoras.length; i++) {
    var filaInicio = filasSeparadoras[i] + 1;
    var filaFin = i + 1 < filasSeparadoras.length ? filasSeparadoras[i + 1] - 1 : ultimaFila;

    if (filaInicio > filaFin) {
      continue;
    }

    ActualizarConteoBloqueInventario_(inventario, filaInicio, filaFin, columnas);
  }
}

function ActualizarConteoBloqueInventario_(inventario, filaInicio, filaFin, columnas) {
  if (filaInicio > filaFin) {
    return;
  }

  var cantidadFilas = filaFin - filaInicio + 1;
  var productos = inventario.getRange(filaInicio, columnas.producto, cantidadFilas, 1).getDisplayValues();
  var conteos = [];
  var numero = 1;

  for (var i = 0; i < productos.length; i++) {
    var producto = String(productos[i][0] || "").trim();

    if (producto === "") {
      conteos.push([""]);
    } else {
      conteos.push([numero]);
      numero++;
    }
  }

  inventario.getRange(filaInicio, columnas.numero, cantidadFilas, 1).setValues(conteos);
}

function EsRenglonSeparadorInventario_(inventario, fila, columnas) {
  columnas = columnas || ObtenerColumnasInventario_(inventario);

  if (fila < 4) {
    return false;
  }

  var combinados = inventario.getRange(fila, columnas.producto).getMergedRanges();

  for (var i = 0; i < combinados.length; i++) {
    if (combinados[i].getRow() === fila && combinados[i].getNumColumns() > 1) {
      return true;
    }
  }

  return false;
}

function ReiniciarFilasEnBloques_(inventario, filas) {
  if (filas.length === 0) {
    return;
  }

  var inicioBloque = filas[0];
  var filaAnterior = filas[0];

  for (var i = 1; i < filas.length; i++) {
    var filaActual = filas[i];

    if (filaActual === filaAnterior + 1) {
      filaAnterior = filaActual;
      continue;
    }

    ReiniciarMovimientosInventario_(inventario, inicioBloque, filaAnterior - inicioBloque + 1);
    inicioBloque = filaActual;
    filaAnterior = filaActual;
  }

  ReiniciarMovimientosInventario_(inventario, inicioBloque, filaAnterior - inicioBloque + 1);
}

function ObtenerHojaObligatoria_(ss, nombreHoja) {
  var hoja = ss.getSheetByName(nombreHoja);

  if (!hoja) {
    throw new Error('La hoja "' + nombreHoja + '" no existe.');
  }

  return hoja;
}

function ObtenerValorPorEtiqueta_(hoja, etiqueta) {
  var celdaEtiqueta = BuscarCeldaPorEtiqueta_(hoja, etiqueta);

  if (!celdaEtiqueta) {
    throw new Error('No se encontro la etiqueta "' + etiqueta + '" en la hoja "' + hoja.getName() + '".');
  }

  var candidatos = ObtenerCeldasDeEntrada_(hoja, celdaEtiqueta);

  for (var i = 0; i < candidatos.length; i++) {
    var valor = String(candidatos[i].getDisplayValue()).trim();

    if (valor !== "") {
      return valor;
    }
  }

  return "";
}

function LimpiarValorPorEtiqueta_(hoja, etiqueta) {
  var celdaEtiqueta = BuscarCeldaPorEtiqueta_(hoja, etiqueta);

  if (!celdaEtiqueta) {
    return;
  }

  var candidatos = ObtenerCeldasDeEntrada_(hoja, celdaEtiqueta);

  for (var i = 0; i < candidatos.length; i++) {
    var rango = ObtenerRangoCompletoSiEstaCombinado_(candidatos[i]);

    if (!rango.getFormula()) {
      rango.clearContent();
      return;
    }
  }
}

function BuscarCeldaPorEtiqueta_(hoja, etiqueta) {
  var valores = hoja.getRange(1, 1, Math.min(12, hoja.getMaxRows()), Math.min(8, hoja.getMaxColumns())).getDisplayValues();
  var etiquetaNormalizada = NormalizarTexto_(etiqueta);

  for (var fila = 0; fila < valores.length; fila++) {
    for (var columna = 0; columna < valores[fila].length; columna++) {
      var valorNormalizado = NormalizarTexto_(valores[fila][columna]);

      if (valorNormalizado === etiquetaNormalizada || valorNormalizado === etiquetaNormalizada + ":") {
        return hoja.getRange(fila + 1, columna + 1);
      }
    }
  }

  return null;
}

function ObtenerCeldasDeEntrada_(hoja, celdaEtiqueta) {
  var fila = celdaEtiqueta.getRow();
  var columna = celdaEtiqueta.getColumn();
  var candidatos = [];

  for (var offsetFila = 0; offsetFila <= 1; offsetFila++) {
    for (var offsetColumna = 1; offsetColumna <= 3; offsetColumna++) {
      var filaCandidata = fila + offsetFila;
      var columnaCandidata = columna + offsetColumna;

      if (filaCandidata <= hoja.getMaxRows() && columnaCandidata <= hoja.getMaxColumns()) {
        candidatos.push(hoja.getRange(filaCandidata, columnaCandidata));
      }
    }
  }

  return candidatos;
}

function ObtenerRangoCompletoSiEstaCombinado_(rango) {
  var combinados = rango.getMergedRanges();
  return combinados.length ? combinados[0] : rango;
}

function BuscarProductoInventario_(inventario, producto, columnas) {
  var ultimaFila = inventario.getLastRow();

  if (ultimaFila < 4) {
    return null;
  }

  columnas = columnas || ObtenerColumnasInventario_(inventario);
  var ultimaColumna = inventario.getLastColumn();
  var datos = inventario.getRange(4, 1, ultimaFila - 3, ultimaColumna).getValues();
  var productoBuscado = NormalizarTexto_(producto);

  for (var i = 0; i < datos.length; i++) {
    var filaInventario = i + 4;
    var nombreProducto = String(datos[i][columnas.producto - 1] || "").trim();

    if (EsRenglonSeparadorInventario_(inventario, filaInventario, columnas)) {
      continue;
    }

    if (nombreProducto !== "" && NormalizarTexto_(nombreProducto) === productoBuscado) {
      var existencia = parseFloat(datos[i][columnas.existencia - 1]);

      return {
        fila: filaInventario,
        numero: datos[i][columnas.numero - 1],
        producto: nombreProducto,
        medida: datos[i][columnas.medida - 1],
        existencia: isNaN(existencia) ? 0 : existencia
      };
    }
  }

  return null;
}

function ObtenerColumnasInformacionProductoInventario_(inventario, columnas) {
  columnas = columnas || ObtenerColumnasInventario_(inventario);

  var columnasControl = [
    columnas.stockLunesJueves,
    columnas.stockViernesDomingo,
    columnas.entrada,
    columnas.salida,
    columnas.decomiso,
    columnas.existencia
  ];
  var primeraColumnaControl = Math.min.apply(null, columnasControl);
  var columnasInformacion = [];
  var columnasOrdenadas = [];

  for (var columna = columnas.producto; columna < primeraColumnaControl; columna++) {
    columnasInformacion.push(columna);
  }

  if (columnasInformacion.length === 0) {
    columnasInformacion.push(columnas.producto);
  }

  AgregarColumnaUnica_(columnasOrdenadas, columnas.producto);
  AgregarColumnaUnica_(columnasOrdenadas, columnas.suplidor);

  for (var i = 0; i < columnasInformacion.length; i++) {
    AgregarColumnaUnica_(columnasOrdenadas, columnasInformacion[i]);
  }

  return columnasOrdenadas;
}

function AgregarColumnaUnica_(columnas, columna) {
  if (!columna || columnas.indexOf(columna) !== -1) {
    return;
  }

  columnas.push(columna);
}

function ObtenerEncabezadosColumnasInventario_(inventario, columnas) {
  var columnasInventario = ObtenerColumnasInventario_(inventario);
  var maxFilasEncabezado = Math.min(3, inventario.getMaxRows());
  var maxColumnas = inventario.getLastColumn();
  var encabezados = inventario.getRange(1, 1, maxFilasEncabezado, maxColumnas).getDisplayValues();
  var nombres = [];

  for (var i = 0; i < columnas.length; i++) {
    var columna = columnas[i];
    var nombre = "";

    for (var fila = encabezados.length - 1; fila >= 0; fila--) {
      nombre = String(encabezados[fila][columna - 1] || "").trim();

      if (nombre !== "") {
        break;
      }
    }

    if (nombre === "") {
      if (columna === columnasInventario.producto) {
        nombre = "Producto";
      } else if (columna === columnasInventario.suplidor) {
        nombre = "Distribuidor";
      } else if (columna === columnasInventario.medida) {
        nombre = "Medida";
      } else {
        nombre = "Informacion " + (i + 1);
      }
    }

    nombres.push(nombre);
  }

  return nombres;
}

function ObtenerColumnasInventario_(inventario) {
  var maxFilasEncabezado = Math.min(5, inventario.getMaxRows());
  var maxColumnas = inventario.getLastColumn();
  var encabezados = inventario.getRange(1, 1, maxFilasEncabezado, maxColumnas).getDisplayValues();
  var columnas = {
    numero: 1,
    producto: 2,
    medida: 3,
    suplidor: 4,
    stockLunesJueves: 5,
    stockViernesDomingo: 6,
    entrada: 7,
    salida: 8,
    decomiso: 9,
    existencia: 10
  };

  for (var fila = 0; fila < encabezados.length; fila++) {
    for (var columna = 0; columna < encabezados[fila].length; columna++) {
      var texto = NormalizarTexto_(encabezados[fila][columna]);

      if (texto === "numero en inventario") {
        columnas.numero = columna + 1;
      } else if (texto === "nombre del producto") {
        columnas.producto = columna + 1;
      } else if (texto === "medida") {
        columnas.medida = columna + 1;
      } else if (texto === "suplidor" || texto === "proveedor" || texto === "distribuidor") {
        columnas.suplidor = columna + 1;
      } else if (texto === "l,m,m,j" || texto === "lmmj") {
        columnas.stockLunesJueves = columna + 1;
      } else if (texto === "v,s,d" || texto === "vsd") {
        columnas.stockViernesDomingo = columna + 1;
      } else if (texto === "entradas") {
        columnas.entrada = columna + 1;
      } else if (texto === "salidas") {
        columnas.salida = columna + 1;
      } else if (texto === "decomiso") {
        columnas.decomiso = columna + 1;
      } else if (texto === "inventario automatico") {
        columnas.existencia = columna + 1;
      }
    }
  }

  return columnas;
}

function NormalizarTexto_(texto) {
  return String(texto || "")
    .trim()
    .toLowerCase()
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "")
    .replace(/:/g, "");
}

function ConvertirNumero_(valor) {
  var texto = String(valor || "").trim().replace(",", ".");

  if (texto === "") {
    return NaN;
  }

  var numero = parseFloat(texto);

  return isNaN(numero) ? NaN : numero;
}

function ObtenerProductosMiseAnPlace() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var hoja = ss.getSheetByName("MISE AN PLACE");

  if (!hoja) {
    throw new Error('La hoja "MISE AN PLACE" no existe.');
  }

  var ultimaFila = hoja.getLastRow();

  if (ultimaFila < 2) {
    return [];
  }

  var datos = hoja.getRange(2, 2, ultimaFila - 1, 2).getValues();
  var productos = [];

  for (var i = 0; i < datos.length; i++) {
    var producto = String(datos[i][0]).trim();
    var cantidad = parseFloat(datos[i][1]);

    if (producto !== "") {
      productos.push({
        producto: producto,
        cantidad: isNaN(cantidad) ? 0 : cantidad
      });
    }
  }

  return productos;
}

// REGISTRAR DECOMISO MISE AN PLACE
// ===================================

function RegistrarDecomisoMiseAnPlace(data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var mise = ss.getSheetByName("MISE AN PLACE");
  var historial = ss.getSheetByName("HISTORIAL DECOMISO COCINA");

  if (!mise) {
    throw new Error('La hoja "MISE AN PLACE" no existe.');
  }

  if (!historial) {
    throw new Error('La hoja "HISTORIAL DECOMISO COCINA" no existe.');
  }

  var producto = String(data.producto || "").trim();
  var fecha = String(data.fecha || "").trim();
  var cantidad = parseFloat(String(data.cantidad || "").replace(",", "."));
  var motivo = String(data.motivo || "").trim();
  var area = String(data.area || "").trim();
  var responsable = String(data.responsable || "").trim();

  if (producto === "") {
    throw new Error("Debe ingresar el producto.");
  }

  if (fecha === "") {
    throw new Error("Debe ingresar la fecha.");
  }

  if (isNaN(cantidad) || cantidad <= 0) {
    throw new Error("Debe ingresar una cantidad válida mayor a cero.");
  }

  if (motivo === "") {
    throw new Error("Debe ingresar la razón o motivo.");
  }

  if (area === "") {
    throw new Error("Debe ingresar el área.");
  }

  if (responsable === "") {
    throw new Error("Debe ingresar el responsable.");
  }

  var lock = ObtenerBloqueoDocumento_();

  try {
    var ultimaFila = mise.getLastRow();

    if (ultimaFila < 2) {
      throw new Error('No hay productos registrados en "MISE AN PLACE".');
    }

    var datos = mise.getRange(2, 2, ultimaFila - 1, 2).getValues();

    var filaProducto = -1;
    var existenciaActual = 0;

    for (var i = 0; i < datos.length; i++) {
      var productoHoja = String(datos[i][0]).trim();

      if (productoHoja.toLowerCase() === producto.toLowerCase()) {
        filaProducto = i + 2;
        existenciaActual = parseFloat(datos[i][1]);

        if (isNaN(existenciaActual)) {
          existenciaActual = 0;
        }

        break;
      }
    }

    if (filaProducto === -1) {
      throw new Error('El producto "' + producto + '" no existe en MISE AN PLACE.');
    }

    if (cantidad > existenciaActual) {
      throw new Error(
        "La cantidad a decomisar excede la existencia actual.\n\n" +
        "Producto: " + producto + "\n" +
        "Existencia actual: " + existenciaActual + "\n" +
        "Cantidad solicitada: " + cantidad
      );
    }

    var nuevaExistencia = existenciaActual - cantidad;

    // Reduce la cantidad en MISE AN PLACE, columna C.
    mise.getRange(filaProducto, 3).setValue(nuevaExistencia);

    // Guarda en HISTORIAL DECOMISO COCINA:
    // A Producto, B Fecha, C Cantidad, D Motivo, E Area, F Responsable.
    historial.getRange(historial.getLastRow() + 1, 1, 1, 6).setValues([[
      producto,
      fecha,
      cantidad,
      motivo,
      area,
      responsable
    ]]);

    SpreadsheetApp.flush();

    return {
      ok: true,
      mensaje: "Decomiso registrado correctamente.\n\nNueva existencia de " + producto + ": " + nuevaExistencia
    };
  } finally {
    LiberarBloqueo_(lock);
  }
}

function RegistrarSalidaRapidaMiseAnPlace(items) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var hoja = ss.getSheetByName("MISE AN PLACE");

  if (!hoja) {
    throw new Error('La hoja "MISE AN PLACE" no existe.');
  }

  if (!items || items.length === 0) {
    throw new Error("Debe ingresar al menos un producto.");
  }

  var ultimaFila = hoja.getLastRow();

  if (ultimaFila < 5) {
    throw new Error('No hay productos registrados en "MISE AN PLACE".');
  }

  var rango = hoja.getRange(5, 2, ultimaFila - 4, 5);
  var datos = rango.getValues();
  var mapa = {};

  for (var i = 0; i < datos.length; i++) {
    var productoHoja = String(datos[i][0] || "").trim();

    if (productoHoja !== "") {
      mapa[NormalizarTexto_(productoHoja)] = {
        indice: i,
        producto: productoHoja,
        cantidad: parseFloat(datos[i][1])
      };
    }
  }

  var acumulados = {};

  items.forEach(function (item) {
    var producto = String(item.producto || "").trim();
    var cantidad = parseFloat(String(item.cantidad || "").replace(",", "."));

    if (producto === "") {
      throw new Error("Hay una fila sin producto.");
    }

    if (isNaN(cantidad) || cantidad <= 0) {
      throw new Error('La cantidad de "' + producto + '" debe ser mayor a cero.');
    }

    var clave = NormalizarTexto_(producto);

    if (!mapa[clave]) {
      throw new Error('El producto "' + producto + '" no existe en MISE AN PLACE.');
    }

    acumulados[clave] = (acumulados[clave] || 0) + cantidad;
  });

  Object.keys(acumulados).forEach(function (clave) {
    var info = mapa[clave];
    var existenciaActual = isNaN(info.cantidad) ? 0 : info.cantidad;
    var cantidadSalida = acumulados[clave];

    if (cantidadSalida > existenciaActual) {
      throw new Error(
        "La salida excede la existencia actual.\n\n" +
        "Producto: " + info.producto + "\n" +
        "Existencia actual: " + existenciaActual + "\n" +
        "Cantidad solicitada: " + cantidadSalida
      );
    }

    datos[info.indice][1] = existenciaActual - cantidadSalida;
    datos[info.indice][4] = existenciaActual - cantidadSalida;
  });

  rango.setValues(datos);
  SpreadsheetApp.flush();

  return {
    ok: true,
    mensaje: "Salida rapida registrada correctamente.\n\nProductos actualizados: " + Object.keys(acumulados).length
  };
}
